git status   查看我当前的仓库的状态
git add file（文件名） 把这个文件追踪了
git add .     把更新的所有文件都更新追踪了
git commit  -m "增加的测试文件"   注释本次提交的内容 
git push  提交到github远程仓库
git pull  拉取远程的代码
git branch  查看本地分支
git branch -a   查看所有远程端的分支
git checkout -b branch1（分支名字）   创建分支
git push --set-upstream origin branch1   把分支推送到github上面
git checkout master（分支名）   切换工作分支
git branch -d branch1 （要删除的分支名字）   删除本地分支
git branch -r -d origin/branch1 （要删除的分支名字）       删除远程分支,此时github上没有删除
git push origin :branch1    （要删除的分支名字）    origin后面一定要加空格再加冒号，推送上去删除完毕
git merge mergedemo （要合并的分支名）    合并分支
git reset --hard HEAD^       版本回退，提交错内容可以用该命令进行回退  多加一个^  就多回退一个版本
git reset --hard HEAD~100    回退100个版本
git reflog     查看所有提交版本的id
git reset --hard  版本号id   就回到我们这个版本号    回退到指定版本
